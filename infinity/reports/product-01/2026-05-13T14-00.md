### 2. API Route 생성

**파일**: `src/app/api/score/route.ts`

전체 구현은 `infinity/artifacts/product-01/ai-scoring-prompt.md`의 route.ts 스케치 기반.
추가할 내용:

```typescript
import Anthropic from "@anthropic-ai/sdk";
import { z } from "zod";
import { NextRequest } from "next/server";

const ScoreSchema = z.object({
  score: z.number().int().min(0).max(10),
  comment: z.string().max(80),
  tags: z.array(z.string()).max(2),
});

const SYSTEM_PROMPT = `당신은 "덕력 채점관"입니다...`; // ai-scoring-prompt.md에서 가져옴

export async function POST(req: NextRequest) {
  try {
    const { imageBase64, mimeType, memo, toneMode = "soft" } = await req.json();

    if (!imageBase64 || !mimeType) {
      return Response.json({ error: "이미지가 필요합니다" }, { status: 400 });
    }

    const client = new Anthropic(); // ANTHROPIC_API_KEY 환경변수 자동 사용

    const response = await client.messages.create({
      model: "claude-sonnet-4-6",
      max_tokens: 256,
      system: SYSTEM_PROMPT.replace("{{toneMode}}", toneMode),
      messages: [{
        role: "user",
        content: [
          { type: "image", source: { type: "base64", media_type: mimeType, data: imageBase64 } },
          ...(memo ? [{ type: "text" as const, text: memo }] : []),
        ],
      }],
    });

    const raw = response.content[0].type === "text" ? response.content[0].text : "{}";
    // JSON 파싱 — 모델이 markdown 코드블록으로 감쌀 수 있으므로 추출
    const jsonMatch = raw.match(/\{[\s\S]*\}/);
    if (!jsonMatch) return Response.json({ error: "채점 실패" }, { status: 500 });

    const parsed = ScoreSchema.parse(JSON.parse(jsonMatch[0]));
    return Response.json(parsed);
  } catch (e) {
    console.error("[score API]", e);
    return Response.json({ error: "채점 중 오류가 발생했습니다" }, { status: 500 });
  }
}
```

const res = await fetch("/api/score", {
  method: "POST",
  headers: { "Content-Type": "application/json" },
  body: JSON.stringify({
    imageBase64: /* FileReader로 변환한 base64 */,
    mimeType: file.type,

- `File → base64` 변환: `FileReader.readAsDataURL` 후 prefix 제거 (`data:image/jpeg;base64,` 제거)
